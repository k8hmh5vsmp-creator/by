
    A[研究目标：基于国产多光谱影像结合机器学习的森林生物量估算] --> B{数据准备与预处理};

    B --> B1[遥感数据获取: 国产GF-1/GF-6多光谱 & GF-5高光谱影像];
    B --> B2[地面样地数据整理: 森林AGB实测数据];
    B2 --> B2_1[数据划分: 训练集(70%) & 测试集(30%)];

    B1 & B2 --> C{影像精细化预处理};
    C --> C1[辐射定标];
    C --> C2[大气校正 (FLAASH)];
    C --> C3[几何校正];
    C --> C4[地形校正 (结合DEM数据)];

    C --> D{多维度特征提取与优选};
    D --> D1[光谱与指数特征提取];
    D1 --> D1_1[原始波段反射率];
    D1 --> D1_2[植被指数: NDVI, EVI, SAVI等];
    D1 --> D1_3[高光谱红边特征: 红边位置/幅值等];
    D --> D2[纹理特征提取];
    D2 --> D2_1[GLCM均值、方差、对比度、熵 (多窗口尺度)];
    D --> D3[地形特征提取: 海拔、坡度、坡向 (基于DEM)];

    D --> D4[特征降维与优选];
    D4 --> D4_1[Pearson相关性分析 (初步筛选)];
    D4 --> D4_2[基于树模型特征重要性评估 或 RFE (精选最优特征子集)];

    D4 --> E{机器学习模型构建与超参数调优};
    E --> E1[基准模型构建: 多元线性回归 (MLR) / 支持向量回归 (SVR)];
    E --> E2[集成学习模型构建 (核心)];
    E2 --> E2_1[随机森林 (RandomForest)];
    E2 --> E2_2[XGBoost];
    E --> E3[超参数自适应优化: GridSearchCV 或 贝叶斯优化];

    E --> F{模型评价与生物量制图};
    F --> F1[精度验证 (使用独立测试集)];
    F1 --> F1_1[评价指标: R^2, RMSE, rRMSE];
    F --> F2[反演制图: 全区森林生物量预测];
    F2 --> F2_1[输出: “森林地上生物量空间分布图”];

    F --> G[研究成果与论文撰写];
